    <script>
      const form = document.getElementById('feedbackForm');
      
      form.onsubmit = function(e) {
        e.preventDefault();
        const btn = form.querySelector('button');
        btn.disabled = true;
        btn.innerText = "A enviar...";
        
        // Captura os dados do formulário manualmente para evitar erros de conversão
        const formData = new FormData(form);
        const dataObj = {};
        formData.forEach((value, key) => {
          dataObj[key] = value;
        });

        google.script.run
          .withSuccessHandler(msg => {
            document.querySelector('.container').innerHTML = `
              <div class="text-center py-5">
                <h2 style="color: #28a745;">✅</h2>
                <h4>${msg}</h4>
              </div>`;
          })
          .withFailureHandler(err => {
            document.getElementById('status').innerHTML = `<span class="text-danger">Erro: ${err}</span>`;
            btn.disabled = false;
            btn.innerText = "Finalizar Avaliação";
          })
          .processarForm(dataObj); // Passamos o objeto limpo aqui
      };
    </script>
